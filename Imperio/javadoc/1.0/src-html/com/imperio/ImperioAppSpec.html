024package com.imperio;
025
026/**
027 * Instances of this class contain all the necessary information to generate an
028 * Imperio application instance.
029 * 
030 * @author afelsher
031 *
032 * @since 1.0.0
033 */
034public class ImperioAppSpec {
035
036    public static final String DEFAULT_AUTHOR = null;
037    public static final String DEFAULT_COPYRIGHT = null;
038    public static final String DEFAULT_DESCRIPTION = null;
039    public static final ErrorHandler DEFAULT_ERROR_HANDLER =
040            new ErrorHandlerException();
041    public static final String DEFAULT_HELP_ADDENDA = null;
042    public static final HelpGen DEFAULT_HELP_GEN = new HelpGenDefault();
043    public static final OptionHelp DEFAULT_HELP_OPT;
044    public static final String DEFAULT_INVOCATION = null;
045    public static final ImpLogger DEFAULT_LOGGER = new ImpLoggerNull();
046    public static final boolean DEFAULT_NO_MATCH_ERROR = true;
047    public static final String DEFAULT_SEE_ALSO = null;
048    public static final String DEFAULT_USAGE_ADDENDA = null;
049    public static final UsageGen DEFAULT_USAGE_GEN = new UsageGenNames();
050
051    static {
052        OptionSpec spec = OptionHelp.generateSpec();
053        spec.auto = true;
054        DEFAULT_HELP_OPT = OptionHelp.generate(spec);
055    }
056
057    public String author = DEFAULT_AUTHOR;
058    public String copyright = DEFAULT_COPYRIGHT;
059    public String description = DEFAULT_DESCRIPTION;
060    public ErrorHandler errorHandler = DEFAULT_ERROR_HANDLER;
061    public String helpAddenda = DEFAULT_HELP_ADDENDA;
062    public HelpGen helpGen = DEFAULT_HELP_GEN;
063    public OptionHelp helpOpt = DEFAULT_HELP_OPT;
064    public String invocation;
065    public ImpLogger logger = DEFAULT_LOGGER;
066    public boolean unknownOptionError = DEFAULT_NO_MATCH_ERROR;
067    public String seeAlso = DEFAULT_SEE_ALSO;
068    public String usageAddenda = DEFAULT_USAGE_ADDENDA;
069    public UsageGen usageGen = DEFAULT_USAGE_GEN;
070
071    /**
072     * @since 1.0.0
073     */
074    public ImperioAppSpec() {
075        StackTraceElement[] ste = new Throwable().getStackTrace();
076        String cn = ste[ste.length - 1].getClassName();
077        int index = cn.lastIndexOf(".");
078        if (index > 0) {
079            index++;
080        } else {
081            index = 0;
082        }
083        invocation = cn.substring(index);
084    }
085
086    /**
087     * @param spec
088     *            app spec to duplicate
089     * 
090     * @since 1.0.0
091     */
092    public ImperioAppSpec(ImperioAppSpec spec) {
093        this.author = spec.author;
094        this.copyright = spec.copyright;
095        this.description = spec.description;
096        this.errorHandler = spec.errorHandler;
097        this.helpAddenda = spec.helpAddenda;
098        this.helpGen = spec.helpGen;
099        this.helpOpt = spec.helpOpt;
100        this.invocation = spec.invocation;
101        this.logger = spec.logger;
102        this.seeAlso = spec.seeAlso;
103        this.unknownOptionError = spec.unknownOptionError;
104        this.usageAddenda = spec.usageAddenda;
105        this.usageGen = spec.usageGen;
106    }
107
108}
