036/**
037 * <p>
038 * Instances of this class represent an application and can parse a CLI
039 * invocation of the application.
040 * </p>
041 * 
042 * <p>
043 * <b>Specifying CLI Options:</b>
044 * </p>
045 * 
046 * <p>
047 * The most unambiguous way to specify a option is to use "--[opt-name]", where
048 * "[opt-name]" is the full option name (e.g. "verbose"). If an arg begins with
049 * "--" (e.g. "--verbose"), we will compare the arg (sans "--") to the full
050 * option name.
051 * </p>
052 * 
053 * <p>
054 * For example, consider an application with only one valid option: "alice".
055 * </p>
056 * 
057 * <p>
058 * This Alice option could be specified with "--alice".
059 * </p>
060 * 
061 * <p>
062 * In addition to specifying a option via the "--[opt-name]" form, you can also
063 * specify via the single-character POSIX "-[char]" form. Some notes on POSIX
064 * options:
065 * </p>
066 * <ol>
067 * <li>Each option is only linked to a single character. There are no "alias"
068 * characters unless the application codes multiple options to perform the same
069 * function, which we do not recommend.</li>
070 * <li>The char is explicitly set by the application, and need not bear any
071 * relation to the option name. For example, a "verbose" option could be coded
072 * to match with "-v", but also "-z", "-q", or whatever the application
073 * developer desires.</li>
074 * <li>For the sake of user-friendliness, we recommend making the single
075 * character derive from the option name in some form, if possible (for example:
076 * make "-v" equivalent to "--verbose").</li>
077 * <li>Out of a desire to support applications with very large varieties of
078 * parameters, the application developer is <b>not</b> required to specify a
079 * single character, so an application may have options which must be specified
080 * via the full option name ("--[opt-name]").</li>
081 * <li>The application developer is similarly <b>not</b> required to specify a
082 * name for the option. However, the application developer <b>must</b> specify
083 * either a name or a single character or both.</li>
084 * </ol>
085 * 
086 * <p>
087 * Assuming the application developer does nothing to substantially modify the
088 * behavior via callbacks and such, options are mostly POSIX-compliant, meaning:
089 * </p>
090 * <ol>
091 * <li>An option is a hyphen followed by a single alphanumeric character, like
092 * this: "-o".</li>
093 * <li>An option may require an argument (which must appear immediately after
094 * the option); for example, "-o [argument]".</li>
095 * <li>Options that do not require arguments can be grouped after a hyphen, so,
096 * for example, "-lst" is equivalent to "-t -l -s".</li>
097 * <li>Options can appear in any order; thus "-lst" is equivalent to
098 * "-tls".</li>
099 * <li>Options can appear multiple times.</li>
100 * <li>Options precede other non-option arguments: "-lst nonoption".</li>
101 * <li>The -- argument terminates options.</li>
102 * </ol>
103 * 
104 * <p>
105 * Options deviate from POSIX in these ways:
106 * </p>
107 * <ol>
108 * <li>Options with arguments must be separated from their arguments. "-o
109 * [argument]" is valid. "-o[argument]" is not.</li>
110 * <li>The - argument is not used to represent anything. It is ignored by this
111 * library.</li>
112 * <li>The -- string can be used to specify long-form options. However, when
113 * used by itself it does terminate the options to be parsed per POSIX.</li>
114 * </ol>
115 * 
116 * <p>
117 * For example, consider an application with only two valid options: "alice" and
118 * "bob". Assume the application developer linked the "x" char to "alice" and
119 * didn't link any char to "bob".
120 * </p>
121 * 
122 * <p>
123 * Alice could be specified with "-x" or "--alice", but <b>not</b> "-a".
124 * </p>
125 * 
126 * <p>
127 * Bob could be specified with "--bob", but <b>not</b> "-b".
128 * </p>
129 * 
130 * <p>
131 * For the sake of brevity, multiple single-character options can be specified
132 * in a single arg <b>if</b> (and only if) all the options except the last one
133 * do not specify values.
134 * </p>
135 * 
136 * <p>
137 * For example, consider an application with only three valid options: "alice",
138 * "bob", and "charlie". "charlie" takes a value, but the other options are
139 * flags. Assume the application developer linked the "a" char to "alice", the
140 * "b" char to "bob", and the "c" char to "charlie".
141 * </p>
142 * 
143 * <p>
144 * "-a" would specify Alice.
145 * </p>
146 * 
147 * <p>
148 * "-aa" would specify Alice twice.
149 * </p>
150 * 
151 * <p>
152 * "-b" would specify Bob.
153 * </p>
154 * 
155 * <p>
156 * "-ab" would specify Alice and Bob.
157 * </p>
158 * 
159 * <p>
160 * "-aab" would specify Alice twice and Bob once.
161 * </p>
162 * 
163 * <p>
164 * "-c mycharlie" would specify the value "mycharlie" for Charlie.
165 * </p>
166 * 
167 * <p>
168 * "-abc mycharlie" would specify Alice and Bob once, and would specify the
169 * value "mycharlie" for Charlie
170 * </p>
171 * 
172 * <p>
173 * "-acb" and "-acb mycharlie" would be invalid
174 * </p>
175 *
176 * @since 1.0.0
177 */

279    /**
280     * @param option
281     *            new option
282     * 
283     * @throws InternalImperioException
284     *             if an error occurs
285     * 
286     * @since 1.0.0
287     */
288    public void addOption(Option option) throws InternalImperioException {
289        Character ch = option.character;
290        String name = option.name;
291        OptionIndex index = null;
292
293        try {
294            index = new OptionIndex(name, ch);
295
296            if (containsOption(name)) {
297                errorHandler.err(ErrorType.GENERIC, this,
298                        "Multiple options%s use option name \'%s\'",
299                        (option.auto
300                                ? " (including at least one option auto-"
301                                        + "generated by the option library"
302                                : ""),
303                        name);
304                return;
305            }
306
307            if (containsOption(ch)) {
308                errorHandler.err(ErrorType.GENERIC, this,
309                        "Multiple options%s use option char \'%s\'",
310                        (option.auto
311                                ? " (including at least one option auto-"
312                                        + "generated by the option library"
313                                : ""),
314                        ch);
315                return;
316            }
317        } catch (ImperioException e) {
318            errorHandler.err(ErrorType.IMPERIO, this,
319                    new InternalImperioException(e));
320            return;
321        }
322
323        if (name == null && ch == null) {
324            errorHandler.err(ErrorType.GENERIC, this,
325                    "Option must have a name or a character.");
326        }
327
328        logger.log("");
329        logger.log("Adding   option %s", option.getInvocation());
330        optionsByIndeces.put(index, option);
331
332        if (name != null) {
333            boolean added = false;
334            logger.log("Checking option name %s", name);
335            for (int i = 0; i < optionNames.size(); i++) {
336                if (name.compareTo(optionNames.get(i).name) < 0) {
337                    logger.log("Adding option name %s", name);
338                    optionNames.add(i, index);
339                    added = true;
340                    break;
341                }
342            }
343            if (!added) {
344                logger.log("Adding   option name %s", name);
345                optionNames.add(index);
346            }
347        }
348        if (ch != null) {
349            boolean added = false;
350            logger.log("Checking option char %c", ch);
351            for (int i = 0; i < optionChars.size(); i++) {
352                if (ch < optionChars.get(i).character) {
353                    logger.log("Adding   option char %c", ch);
354                    optionChars.add(i, index);
355                    added = true;
356                    break;
357                }
358            }
359            if (!added) {
360                logger.log("Adding   option char %c", ch);
361                optionChars.add(index);
362            }
363        }
364
365        boolean added = false;
366        logger.log("Checking option index %s", index);
367        for (int i = 0; i < optionIndeces.size(); i++) {
368            if (index.compareTo(optionIndeces.get(i)) < 0) {
369                logger.log("Adding   option index %s", index);
370                optionIndeces.add(i, index);
371                added = true;
372                break;
373            }
374        }
375        if (!added) {
376            logger.log("Adding   option index %s", index);
377            optionIndeces.add(index);
378        }
379    }

381    /**
382     * Check if the given character is in use.
383     * 
384     * @param character character to check
385     * 
386     * @return {@code true} if this app contains an option assigned to the given
387     *         character, else {@code false}
388     * 
389     * @throws ImperioException if the given character is null
390     * 
391     * @since 1.0.0
392     */
393    public boolean containsOption(Character character) throws ImperioException {
394        if ((character != null)) {
395            Iterator<OptionIndex> iter = getOptionIndeces();
396            while (iter.hasNext()) {
397                OptionIndex index = iter.next();
398                if (index.compareTo(character) == 0) {
399                    return true;
400                }
401            }
402        }
403        return false;
404    }

406    /**
407     * Check if the given index maps to an option for this application.
408     * 
409     * @param index option index to check
410     * 
411     * @return {@code true} if this app contains an option assigned to the given
412     *         index, else {@code false}
413     * 
414     * @throws ImperioException
415     *             if an error occurs
416     * 
417     * @since 1.0.0
418     */
419    public boolean containsOption(OptionIndex index) throws ImperioException {
420        /*
421         * We can't just check if the Map contains this index, because it's
422         * a hash map, and the hashes may not match even when the indices do
423         * (e.g. index A contains the name and index B contains both the name
424         * and the character).
425         */
426        if (containsOption(index.name)) {
427            return true;
428        }
429        if (containsOption(index.character)) {
430            return true;
431        }
432        return false;
433    }

594    /**
595     * Process the CLI arguments.
596     * 
597     * @param args
598     *            the CLI arguments
599     * @param nonOptArgs
600     *            A container whose value is set by this function to an array of
601     *            the leftover non-option arguments
602     * 
603     * @return {@code true} for normal operation if there were no errors.
604     *         {@code false} if there was an error or the help option was used.
605     * 
606     * @throws InternalImperioException
607     *             If an error occurs and the error handler is configured to
608     *             throw exceptions.
609     * 
610     * @since 1.0.0
611     */
612    public boolean processArgs(String[] args, StringArrContainer nonOptArgs)
613            throws InternalImperioException {
614        int argi = 0;
615        boolean printHelp = false;
616        boolean ret = true;
617
618        logger.log("");
619        logger.log("Processing args (%d)", args == null ? 0 : args.length);
620        argloop: for (argi = 0; (args != null)
621                && (argi < args.length); argi++) {
622            String arg = args[argi];
623            logger.log("");
624            logger.log("");
625            logger.log("Processing arg %d/%d: %s", argi + 1, args.length, arg);
626            if (arg.equals("--")) {
627                logger.log("Option args terminated by %s at %d/%d", arg,
628                        argi + 1, args.length);
629                argi++; // Don't include -- in the leftover args.
630                break;
631            } else if (!arg.startsWith("-")) {
632                logger.log("Option args terminated by non-option arg %d/%d: %s",
633                        argi + 1, args.length, arg);
634                break;
635            } else if (arg.equals("-")) {
636                errorHandler.warn(this,
637                        "Hyphen(s) without option at option %d: \'%s\'", argi,
638                        arg);
639                continue argloop;
640            } else if (arg.startsWith("---")) {
641                errorHandler.err(ErrorType.INVALID_OPTION, this,
642                        "Invalid option \'%s\'. Did you mean \'--%s\'?", arg,
643                        arg.replaceAll("^-+", ""));
644                ret = false;
645                continue argloop;
646            } else {
647                logger.log("Processing subargs for arg %d/%d: %s", argi + 1,
648                        args.length, arg);
649
650                boolean isPosix =
651                        (arg.startsWith("-") && !arg.startsWith("--"));
652                String[] subargs = null;
653                String wholeArg = arg.replaceFirst("-+", "");
654                Option wholeOpt = null;
655                String wholeOptStr = null;
656
657                for (int parmi = 0; parmi < optionNames.size(); parmi++) {
658                    OptionIndex index = optionNames.get(parmi);
659                    String parmName = index.name;
660                    logger.log("Comparing arg %d/%d %s to %s", argi + 1,
661                            args.length, wholeArg, parmName);
662                    if (parmName.equals(wholeArg)) {
663                        wholeOpt = optionsByIndeces.get(index);
664                        wholeOptStr = "Did you mean \'"
665                                + wholeOpt.getInvocation() + "\'?";
666                        break;
667                    }
668                }
669
670                if (isPosix) {
671                    subargs = new String[wholeArg.length()];
672                    for (int b = 0; b < subargs.length; b++) {
673                        subargs[b] = wholeArg.substring(b, b + 1);
674                    }
675                } else {
676                    subargs = new String[] { wholeArg };
677                }
678
679                saloop: for (int sai = 0; sai < subargs.length; sai++) {
680                    Option opt = null;
681                    String subarg = subargs[sai];
682                    Object oldVal = null;
683                    Object val = null;
684                    String valStr = null;
685
686                    logger.log("");
687                    logger.log("Processing subarg %d/%d for arg %d/%d: %s",
688                            sai + 1, subargs.length, argi + 1, args.length,
689                            subarg);
690
691                    if (!isPosix) {
692                        opt = wholeOpt;
693                    } else {
694                        for (int parmi = 0; parmi < optionChars.size(); parmi++) {
695                            OptionIndex index = optionChars.get(parmi);
696                            Character parmChar = index.character;
697
698                            logger.log("Comparing subarg %d/%d %s to %c",
699                                    sai + 1, subargs.length, subarg, parmChar);
700                            if ((parmChar != null)
701                                    && (parmChar == subarg.charAt(0))) {
702                                opt = optionsByIndeces.get(index);
703                                if ((opt.type.archetype == OptionArchetype.VALUE)
704                                        && (sai < subargs.length - 1)) {
705                                    StringWriter sw = new StringWriter();
706
707                                    sw.write(String.format(
708                                    "Cannot include option \'%s\' (which "
709                                            + "takes an argument) in "
710                                            + "multi-option argument "
711                                            + "(\'%s\') unless it is the "
712                                            + "final option.",
713                                    subarg, arg));
714                                    sw.write(" Did you mean \'-");
715                                    sw.write(parmChar);
716                                    sw.write(" <arg> -");
717                                    if (sai > 0) {
718                                        sw.write(wholeArg.substring(0, sai));
719                                    }
720                                    if (sai < subargs.length - 1) {
721                                        sw.write(wholeArg.substring(sai + 1));
722                                    }
723                                    sw.write("\'?");
724                                    if (wholeOpt != null) {
725                                        sw.write(" " + wholeOptStr);
726                                    }
727                                    errorHandler.err(ErrorType.OPTION_PARSING,
728                                            this, sw.toString());
729                                    ret = false;
730                                    continue saloop;
731                                }
732                                logger.log(
733                                        "Subarg %d/%d for arg %d/%d (%s) matches "
734                                                + "option %c",
735                                        sai + 1, subargs.length, argi + 1,
736                                        args.length, subarg, opt.character);
737                                break;
738                            }
739                        }
740                    }
741                    if (opt == null) {
742                        if (unknownOptionError) {
743                            errorHandler.err(ErrorType.UNKNOWN_OPTION,
744                                    this,
745                                    "Unknown option \'%s\' in argument \'%s\'.%s",
746                                    subargs[sai], args[argi],
747                                    ((wholeOptStr == null) ? ""
748                                            : (" " + wholeOptStr)));
749                            ret = false;
750                            continue saloop;
751                        } else {
752                            logger.log("Option args terminated by non-option arg %d/%d: %s",
753                                    argi + 1, args.length, arg);
754                            break argloop;
755                        }
756                    }
757
758                    opt.provided = true;
759                    oldVal = opt.getValue();
760
761                    if (opt.deprecated) {
762                        errorHandler.warn(this, "Using deprecated option: %s",
763                                opt.getInvocation());
764                    }
765
766                    try {
767
768                        if (opt.type.archetype == OptionArchetype.VALUE) {
769                            if (args.length <= argi + 1) {
770                                errorHandler.err(ErrorType.MISSING_ARG,
771                                        this,
772                                        "No value provided for \'%s\'",
773                                        args[argi]);
774                                ret = false;
775                                continue saloop;
776                            }
777                            if (args[argi + 1].startsWith("-")) {
778                                errorHandler.warn(this,
779                                        "Value for \'%s\' (\'%s\') begins with "
780                                                + "hyphen. Possible missing "
781                                                + "value.",
782                                        args[argi], args[argi + 1]);
783                            }
784                            val = valStr = args[++argi];
785                        }
786
787                        switch (opt.type) {
788                        case CUSTOM_FLAG:
789                        case FLAG:
790                            val = true;
791                            opt.setValue(val);
792                            if (opt.linkOption != null) {
793                                opt.linkOption.setValue(false);
794                            }
795                            break;
796                        case DECREMENTER:
797                            val = ((Integer) opt.getValue()) - 1;
798                            opt.setValue(val);
799                            if (opt.linkOption != null) {
800                                opt.linkOption.setValue(opt.getValue());
801                            }
802                            break;
803                        case FILE:
804                            val = new File(valStr);
805                            opt.setValue(val);
806                            break;
807                        case INCREMENTER:
808                            val = ((Integer) opt.getValue()) + 1;
809                            opt.setValue(val);
810                            if (opt.linkOption != null) {
811                                opt.linkOption.setValue(opt.getValue());
812                            }
813                            break;
814                        case ARG:
815                            opt.setValue(val);
816                            break;
817                        case CUSTOM_ARG:
818                        default:
819                            break;
820                        }
821                    } catch (ImperioException e) {
822                        errorHandler.err(ErrorType.IMPERIO, this,
823                                new InternalImperioException(e));
824                    }
825                    if (opt.callback != null) {
826                        try {
827                            opt.callback.callback(this, opt, oldVal, val);
828                        } catch (ImperioException e) {
829                            errorHandler.err(ErrorType.GENERIC, this,
830                                    new InternalImperioException(e));
831                        }
832                    }
833                    if (opt.equals(helpOpt)) {
834                        printHelp = true;
835                    }
836                }
837            }
838        }
839
840        if ((nonOptArgs != null) && (args != null) && (argi < args.length)) {
841            logger.log("Generating non-opts array");
842            String[] nonOpts = new String[args.length - argi];
843            for (int i = 0; (i < nonOpts.length); i++) {
844                logger.log("Adding non-opts arg %d: %s", i, args[argi + i]);
845                nonOpts[i] = args[argi + i];
846            }
847            nonOptArgs.arr = nonOpts;
848        }
849
850        if (printHelp) {
851            try {
852                help.printHelp(this);
853            } catch (ImperioException e) {
854                errorHandler.err(ErrorType.GENERIC, this,
855                        new InternalImperioException(e));
856            }
857            ret = false;
858        } else {
859            Iterator<OptionIndex> iter = getOptionIndeces();
860            while (iter.hasNext()) {
861                OptionIndex index = iter.next();
862                Option opt = optionsByIndeces.get(index);
863                if (opt.required && !opt.provided) {
864                    errorHandler.err(ErrorType.MISSING_REQ_OPTION,
865                            this, "Required option not given: %s",
866                            opt.getInvocation());
867                    ret = false;
868                }
869            }
870        }
871
872        return ret;
873    }
